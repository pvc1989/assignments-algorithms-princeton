43  private void findSegments() {
44  if (itsPoints.length < 4)
45  return;
46  final int nPoints = itsPoints.length;
47  double pqSlope, prSlope, psSlope;
48  for (int p = 0; p != nPoints; ++p) {
49  for (int q = p + 1; q != nPoints; ++q) {
50  pqSlope = itsPoints[p].slopeTo(itsPoints[q]);
51  for (int r = q + 1; r != nPoints; ++r) {
52  prSlope = itsPoints[p].slopeTo(itsPoints[r]);
53  if (pqSlope != prSlope) {
54  continue;
55  }
56  for (int s = r + 1; s != nPoints; ++s) {
57  psSlope = itsPoints[p].slopeTo(itsPoints[s]);
58  if (pqSlope == psSlope) {
59  itsLineSegments.addLast(
60  new LineSegment(itsPoints[p], itsPoints[s]));
61  }
62  } // s-loop
63  } // r-loop
