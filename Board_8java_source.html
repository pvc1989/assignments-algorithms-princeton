42  /*
43  Hamming priority function.
44  The number of blocks in the wrong position, a search node with a small
45  number of blocks in the wrong position is close to the goal, and we prefer
46  a search node that have been reached using a small number of moves.
47  */
48  public int hamming() {
49  int count = 0;
50  int goal = 0; // goal value in this position
51  for (int i = 0; i < dim; i++) {
52  for (int j = 0; j < dim; j++) {
53  goal = i * dim + j + 1;
54  if (board[i][j] != goal) {
55  if (IF_DEBUG) StdOut.println(goal);
56  ++count;
57  }
58  }
59  }
60  if (board[dim - 1][dim - 1] == 0)
61  --count;
62  return count;
63  }

65  /*
66  Manhattan priority function.
67  The sum of the Manhattan distances (sum of the vertical and horizontal
68  distance) from the blocks to their goal positions, plus the number of moves
69  made so far to get to the search node.
70  */
71  public int manhattan() {
72  int iGoal = 0; // row-index of its goal position
73  int jGoal = 0; // column-index of ist goal position
74  int dist = 0;
75  for (int i = 0; i < dim; i++) {
76  for (int j = 0; j < dim; j++) {
77  if (board[i][j] != 0) {
78  iGoal = (board[i][j] - 1) / dim;
79  jGoal = (board[i][j] - 1) % dim;
80  dist += (Math.abs(i - iGoal) + Math.abs(j - jGoal));
81  }
82  }
83  }
84  return dist;
85  }

93  public Board twin() {
94  Board twinObj = new Board(board);
95  // get a random integer uniformly in [0, dim * dim).
96  int k1 = StdRandom.uniform(dim * dim);
97  int i1 = k1 / dim;
98  int j1 = k1 % dim;
99  // its value is not 0

102  i1 = k1 / dim;
103  j1 = k1 % dim;
104  }
105  // get another random integer uniformly in [0, dim * dim).
106  int k2 = StdRandom.uniform(dim * dim);
107  int i2 = k2 / dim;
108  int j2 = k2 % dim;
109  // which is different from the 1st one
110  // and its value is not 0
111  while (k1 == k2 || board[i2][j2] == 0) {
112  k2 = StdRandom.uniform(dim * dim);
113  i2 = k2 / dim;
114  j2 = k2 % dim;
115  }
116  twinObj.swap(i1, j1, i2, j2);
117  return twinObj;
118  }
