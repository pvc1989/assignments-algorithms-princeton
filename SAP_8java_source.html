213  private int bfs(DeluxeBFS vTree, DeluxeBFS wTree) {
214  int minLength = INFINITY;
215  int ancestor = -1;
216  while (true) {
217  if (vTree.done() && wTree.done()) {
218  // Both BFS's have finished and no SAP has been found.
219  break;
220  }
221  if (!vTree.done()) {
222  if (vTree.depth() >= minLength) {
223  // minLength cannot decrease
224  break;
225  }
226  for (int x : vTree.newVertices()) {
227  // whether these vertices are in another tree
228  if (wTree.hasPathTo(x)) {
229  int pathLength = vTree.distTo(x) + wTree.distTo(x);
230  // update, if such path length < current min length
231  if (pathLength < minLength) {
232  minLength = pathLength;
233  ancestor = x;
234  }
235  }
236  }
237  vTree.growOneLevel(); // continue growing the tree
238  }
239  if (!wTree.done()) {
240  if (wTree.depth() >= minLength) {
241  // minLength cannot decrease
242  break;
243  }
244  for (int x : wTree.newVertices()) {
245  // whether these vertices are in another tree
246  if (vTree.hasPathTo(x)) {

248  // update, if such path length < current min length
