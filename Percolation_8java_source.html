51  private void checkIndices(int row, int col) {
52  if (row < 1 || row > nRow)

60  private int xyCheckedTo1D(int row, int col) {
61  checkIndices(row, col);
62  // 0 for the top sentinel
63  // 1 for (1, 1), 2 for (1, 2), ..., n for (1, n)
64  // ...
65  // ((n-1)*n + 1) for (n, 1), ..., ((n-1)*n + n) for (n, n)
66  // (n*n + 1) for the bottom sentinel
67  return (row - 1) * nCol + col;
68  }
72  public void open(int row, int col) {
73  int id = xyCheckedTo1D(row, col);
74  if (alreadyOpen[id]) return;
75  // open this site
76  alreadyOpen[id] = true;
77  ++nOpen;
78  // connect to the top sentinel site
79  if (row == 1) {
80  weakConnection.union(id, iTop);
81  strongConnection.union(id, iTop);
82  }
83  // connect to the bottom sentinel site
84  if (row == nRow) {
85  weakConnection.union(id, iBottom);
86  }
87  // connect to neighboring open sites
88  if (row > 1) { // 1. connect to the UPPER site
89  connectToNeighbors(id, row - 1, col);
90  }
91  if (row < nRow) { // 2. connect to the LOWER site
92  connectToNeighbors(id, row + 1, col);
93  }
94  if (col > 1) { // 3. connect to the LEFT site
95  connectToNeighbors(id, row, col - 1);
96  }
97  if (col < nCol) { // 4. connect to the RIGHT site
98  connectToNeighbors(id, row, col + 1);
99  }
100  }
104  private void connectToNeighbors(int id, int row, int col) {
105  int idNeighbor = xyCheckedTo1D(row, col);
106  if (alreadyOpen[idNeighbor]) {
107  weakConnection.union(id, idNeighbor);
108  strongConnection.union(id, idNeighbor);
109  }
110  }
