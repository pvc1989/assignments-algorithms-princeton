424  private void newGame() {
425  if (BOARD_ROWS == 4 && BOARD_COLS == 4) {
426  board = new BoggleBoard();
427  }
428  else {
429  board = new BoggleBoard(BOARD_ROWS, BOARD_COLS);
430  }
431  clock.setForeground(Color.BLACK);
432  entryField.requestFocus();
433  inGame = true;
434  points = 0;
435  scoreLabel.setText("Current Points:" + points);
436  entryField.setEnabled(true);
437 
438  foundWords = new LinkedHashSet<String>();
439 
440  // set display of word lists to be empty
441  foundWordsList.setListData(emptyList);
442  validWordsList.setListData(emptyList);
443  opponentFoundWordsList.setListData(emptyList);
444 
445  bp.setBoard();
446  bp.unhighlightCubes();
447 
448  // all valid words
449  Iterable<String> words = solver.getAllValidWords(board);
450  validWords = new TreeSet<String>();
451  int possiblePoints = 0;
452  for (String s : words) {
453  validWords.add(s);
454  possiblePoints += scoreWord(s);
455  }
456  possiblePointsLabel.setText("Possible Points: " + possiblePoints);
457 
458  // opponent's words
459  opponentFoundWords = new TreeSet<String>();
460  if (gameDifficulty == NURSERY) {
461  for (String word : validWords)
462  if (nurseryDictionary.contains(word))
463  opponentFoundWords.add(word);
464  }
465 
466  else if (gameDifficulty == SHAKESPEARE) {
467  for (String word : validWords)
468  if (shakespeareDictionary.contains(word) && StdRandom.uniform(3) != 0)
469  opponentFoundWords.add(word);
470  }
471 
472  else if (gameDifficulty == ALGORITHMS) {
473  for (String word : validWords)
474  if (algs4Dictionary.contains(word))
475  opponentFoundWords.add(word);
476  }
477 
478  else if (gameDifficulty == HARD) {
479  for (String word : validWords)
480  if (commonDictionary.contains(word) && StdRandom.bernoulli())
481  opponentFoundWords.add(word);
482  }
483 
484  else if (gameDifficulty == IMPOSSIBLE) {
485  for (String word : validWords)
486  if (StdRandom.uniform(4) != 0)
487  opponentFoundWords.add(word);
488  }
489 
490  // opponent's score
491  oppCurScore = 0;
492  for (String word : opponentFoundWords)
493  oppCurScore += scoreWord(word);
494 
495  oppScoreLabel.setText("Opponent's Points: " + oppCurScore);
496  timer.cancel();
497  elapsedTime = -1;
498  timer = new Timer();
499  timer.schedule(new Countdown(), 0, 1000);
500 
501  }

506  private void endGame() {
507 
508  clock.setText("00:00");
509  clock.setForeground(Color.RED);
510  timer.cancel();
511  entryField.setText("");
512  entryField.setEnabled(false);
513 
514  // display list of all valid words
515  validWordsList.setListData(validWords.toArray());
516 
517  // highlight found words by specifying indices of found words
518  int[] indices = new int[foundWords.size()];
519  int i = 0;
520  int n = 0;
521  for (String s : validWords) {
522  if (foundWords.contains(s))
523  indices[i++] = n;
524  n++;
525  }
526  validWordsList.setSelectedIndices(indices);
527  //validWordsList.setEnabled(false);
528 
529  inGame = false;
530 
531  // compute score, discounting words that both players found
532  int playerScore = points;
533  int opponentScore = oppCurScore;
534  for (String s : foundWords) {
535  if (opponentFoundWords.contains(s)) {
536  playerScore -= scoreWord(s);
537  opponentScore -= scoreWord(s);
538  }
539  }
540 
541  // strike out words in user's list that opponent found
542  Object[] list1 = foundWords.toArray();
543  for (int j = 0; j < list1.length; j++) {
544  if (opponentFoundWords.contains(list1[j])) {
545  list1[j] = "<html><strike>" + list1[j] + "</strike></html>";
546  }
547  }
548  foundWordsList.setListData(list1);
549 
550  // strike out words in opponent's list that user found
551  Object[] list2 = opponentFoundWords.toArray();
552  for (int j = 0; j < list2.length; j++) {
553  if (foundWords.contains(list2[j])) {
554  list2[j] = "<html><strike>" + list2[j] + "</strike></html>";
555  }
556  }
557  opponentFoundWordsList.setListData(list2);
558 
559  // display dialog indicating which player won
560  String winnerMessage = "";
561  if (playerScore > opponentScore) winnerMessage = " You win!\n\n";
562  else if (playerScore < opponentScore) winnerMessage = " The computer wins!\n\n";
563  else winnerMessage = " Tie!\n\n";
564  String scoreMessage = " Final score:\n You: " + playerScore + " - Computer: " + opponentScore;
565  JOptionPane.showMessageDialog(this, winnerMessage + scoreMessage, "Game finished", JOptionPane.PLAIN_MESSAGE);
566  }

571  private class Countdown extends TimerTask {

846  private void dfs(String s, int curChar, int pathIndex, int i, int j) {
847  // if the word has already been found
848  // if (foundWord) return;
849  // out of bounds
850  if (i < 0 || j < 0 || i >= BOARD_ROWS || j >= BOARD_COLS) return;
851  // return if entire word is found
852  if (curChar >= s.length()) {
853  foundWord = true;
854  return;
855  }
856  // can't use a cell more than once
857  for (int n = 0; n < path.length; n++) {
858  if (path[n] == (i*BOARD_COLS)+j) return;
859  }
860  // ignore if character if there is a 'Q' with no 'U'
861  if (curChar != 0 && s.charAt(curChar-1) == 'Q' && s.charAt(curChar) != 'U')
862  return;
863  // increment character count if it is a 'U' after a 'Q' and keep searching
864  if (curChar != 0 && s.charAt(curChar-1) == 'Q' && s.charAt(curChar) == 'U')
865  curChar += 1;
866  if (curChar >= s.length()) {
867  foundWord = true;
868  return;
869  }
870  // if it doesn't have the right character
871  if (cubes[(i*BOARD_COLS)+j].getText().charAt(0) != s.charAt(curChar)) {
872  return;
873  }
874  // mark spot and save location;
875  path[pathIndex] = (i*BOARD_COLS)+j;
876  //visited[i][j] = true;
877  // consider all neighbors
878  for (int ii = -1; ii <= 1; ii++)
879  for (int jj = -1; jj <= 1; jj++)
880  if (!foundWord) dfs(s, curChar+1, pathIndex+1, i + ii, j + jj);
881 
882  if (!foundWord) path[curChar] = -1;
883  }
884  }
