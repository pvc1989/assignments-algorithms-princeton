46  private void sort(int lo, int hi, int d) {
47  // 3-way string quicksort
48  if (hi - 1 <= lo) return;

50  int gt = hi - 1; // exclusive lower bound of >
51  int pivot = charAt(index(lo) + d); // pivot
52  int i = lo + 1;
53  while (i <= gt) {
54  int x = charAt(index(i) + d);
55  if (x < pivot) swap(lt++, i++);
56  else if (x > pivot) swap(gt--, i);
57  else ++i;
58  } // loop invariant:
59  // \forall x \in [lo, lt) : x < pivot
60  // \forall x \in [lt, gt] : x = pivot
61  // \forall x \in (gt, hi) : x > pivot
62  ++gt; // after this point:
63  // \forall x \in [lo, lt) : x < pivot
64  // \forall x \in [lt, gt) : x = pivot
65  // \forall x \in [gt, hi) : x > pivot
66  sort(lo, lt, d);
67  if (pivot >= 0) sort(lt, gt, d+1);
