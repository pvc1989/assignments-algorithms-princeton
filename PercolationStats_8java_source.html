21 public class PercolationStats {

26  public PercolationStats(int n, int trials) {
27  if (n <= 0) {
28  throw new IllegalArgumentException("n must be >= 0.");
29  }
30  if (trials < 1) {
31  throw new IllegalArgumentException("trials must be >= 1.");
32  }
33  // perform trials independent experiments on an n-by-n grid
34  final int[] nOpen = new int[trials];
35  for (int k = 0, i, j; k < trials; ++k) {
36  // Initialize all sites to be blocked.
37  Percolation perc = new Percolation(n);
38  // Repeat the following until the system percolates:
39  while (!perc.percolates()) {
40  // uniform(int n) returns a random integer uniformly in [0, n).
41  i = 1 + StdRandom.uniform(n);
42  j = 1 + StdRandom.uniform(n);
43  perc.open(i, j);
44  }
45  nOpen[k] = perc.numberOfOpenSites();
46  }
47  final int nSquare = n * n;
48  itsMean = StdStats.mean(nOpen) / nSquare;
49  itsStddev = StdStats.stddev(nOpen) / nSquare;
50  itsHalf = 1.96 * itsStddev / Math.sqrt(trials);
51  }
52  public double mean() {
53  // sample mean of percolation threshold
54  return itsMean;
55  }
56  public double stddev() {
57  // sample standard deviation of percolation threshold
58  return itsStddev;
59  }
60  public double confidenceLo() {
61  // low endpoint of 95% confidence interval
62  return itsMean - itsHalf;
63  }
64  public double confidenceHi() {
65  // high endpoint of 95% confidence interval
66  return itsMean + itsHalf;
67  }
