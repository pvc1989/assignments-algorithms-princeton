84  private Iterable<String> trivialEliminate(int x) {

101  // non-trivial elimination
102  private Iterable<String> mincutEliminate(int x) {
103  if (itsMincutCache.containsKey(x))
104  return itsMincutCache.get(x);
105  final int most = itsWins[x] + itsLefts[x];
106  final int n = numberOfTeams();
107  final int target = n * (n + 1) / 2; // n + n*(n-1)/2
108  final int source = target + 1;
109  int capacity = 0;
110  FlowNetwork network = new FlowNetwork(source + 1);
111  int g = 0; // id of the next game vertex
112  for (int i = 0; i != n; ++i) {
113  if (i == x) // ignore query team
114  continue;
115  // team -> target
116  network.addEdge(new FlowEdge(i, target, most - itsWins[i]));
117  for (int j = i+1; j != n; ++j) {
118  if (j == x) // ignore query team
119  continue;
120  // game -> team
121  g = itsGameId[i][j];
122  network.addEdge(new FlowEdge(g, i, INFINITY));
123  network.addEdge(new FlowEdge(g, j, INFINITY));
124  // source -> game
125  network.addEdge(new FlowEdge(source, g, itsGameCount[i][j]));
126  capacity += itsGameCount[i][j];
127  }
128  }
129  // mincut
130  FordFulkerson mincut = new FordFulkerson(network, source, target);
131  if (mincut.value() == capacity) {
132  // It is possible to assign winners to all of the remaining games
133  // in such a way that no team wins more games than x.
134  itsMincutCache.put(x, null);
135  return null;
136  }
137  else {
138  LinkedList<String> list = new LinkedList<String>();
139  for (int i = 0; i != n; ++i)
140  if (mincut.inCut(i))
141  list.add(itsIdToName[i]);
142  itsMincutCache.put(x, list);
143  return list;
144  }
145  }
